from flask import Flask, render_template, request, redirect, url_for app = Flask(__name__) # Збереження завдань у пам'яті (можна замінити на базу даних) tasks = {"day": [], "week": [], "month": [], "year": []} @app.route('/') def home(): return render_template('index.html', tasks=tasks) @app.route('/add_task', methods=['POST']) def add_task(): period = request.form.get('period') task = request.form.get('task') if period in tasks and task: tasks[period].append(task) return redirect(url_for('home')) @app.route('/clear_tasks', methods=['POST']) def clear_tasks(): period = request.form.get('period') if period in tasks: tasks[period] = [] return redirect(url_for('home')) if __name__ == '__main__': app.run(debug=True)
